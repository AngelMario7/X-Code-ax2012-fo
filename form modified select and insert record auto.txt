public boolean modified()
{
    boolean ret;
    UserInfo userInfo;
    BattUserWarehouseAccess battRec;

    ret = super();
    battRec = element.datasource().cursor();

    select userInfo
        where userInfo.Id == battRec.UserId;

    if (userInfo.RecId)
    {
        battRec.Name = userInfo.Name;
        element.datasource().refresh();
    }
    else
    {
        battRec.Name = '';
        element.datasource().refresh();
    }

    return ret;
}
